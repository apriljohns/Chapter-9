unsolved <- homicides %>% 
  group_by(city_name, disposition) %>% 
  count() %>% 
  ungroup() %>% 
  pivot_wider(names_from = disposition,
              values_from = n,
              values_fill = 0) %>% 
  mutate(num_unsolved = `Closed without arrest` + `Open/No arrest`,
         total_homicides = `Closed by arrest` + `Closed without arrest` + `Open/No arrest`) %>% 
  select(city_name, num_unsolved, total_homicides)

nested_homicides <- unsolved %>% 
  group_by(city_name) %>% 
  nest() %>% 
  ungroup()

prop_homicides <- nested_homicides %>% 
  mutate(prop_test = map(data,
                         ~ prop.test(.x$num_unsolved,
                                     .x$total_homicides))) %>% 
  mutate(prop_test = map(prop_test, tidy)) %>% 
  unnest(prop_test) %>% 
  unnest(data)
